// Apply saved theme immediately on load (before DOMContentLoaded to avoid flash)
(function () {
    const saved = localStorage.getItem('theme') || 'light';
    document.body.classList.remove('light-theme', 'dark-theme');
    document.body.classList.add(saved + '-theme');
})();

// ---- TAB SWITCHING ----
function showTab(tabName, clickedBtn) {
    document.querySelectorAll('.tab-content').forEach(tab => {
        tab.classList.remove('active');
    });
    document.querySelectorAll('.nav-item').forEach(btn => {
        btn.classList.remove('active');
    });

    document.getElementById(tabName + '-tab').classList.add('active');
    if (clickedBtn) clickedBtn.classList.add('active');
}

// ---- THEME ----
function setTheme(theme) {
    const body = document.body;
    const boxes = document.querySelectorAll('.theme-box');

    boxes.forEach(box => box.classList.remove('active'));

    // Use remove+add instead of replace() to avoid silent failures
    body.classList.remove('light-theme', 'dark-theme');
    body.classList.add(theme === 'dark' ? 'dark-theme' : 'light-theme');
    document.querySelector(`.theme-box.${theme}`).classList.add('active');

    // Persist across pages
    localStorage.setItem('theme', theme);
}

// Sync active theme-box to match saved theme on load
function syncThemeBox() {
    const saved = localStorage.getItem('theme') || 'light';
    document.querySelectorAll('.theme-box').forEach(box => box.classList.remove('active'));
    const activeBox = document.querySelector(`.theme-box.${saved}`);
    if (activeBox) activeBox.classList.add('active');
}

function openDeleteModal() {

function confirmDelete() {

document.addEventListener('DOMContentLoaded', async () => {
    syncThemeBox();
    renderProfileTab();

    const userDropdownBtn = document.getElementById('userDropdownBtn');
    const userDropdownMenu = document.getElementById('userDropdownMenu');

    if (userDropdownBtn && userDropdownMenu) {
        userDropdownBtn.addEventListener('click', (e) => {
            e.stopPropagation();
            userDropdownMenu.classList.toggle('show');
        });

        document.addEventListener('click', (e) => {
            if (!userDropdownBtn.contains(e.target)) {
                userDropdownMenu.classList.remove('show');
            }
        });
    }

    const userNameEl = document.getElementById('userName');
    const textAvatar = document.getElementById('textAvatar');
    const token = localStorage.getItem('token');

    if (!token) {
        if (userNameEl) userNameEl.textContent = 'Guest';
        if (textAvatar) textAvatar.textContent = 'G';
        if (userDropdownMenu) {
            userDropdownMenu.innerHTML = `
                <div class="user-dropdown-header">
                    <div class="avatar">G</div>
                    <div class="user-dropdown-header-info">
                        <div class="user-dropdown-name">Guest User</div>
                        <div class="user-dropdown-email">Not logged in</div>
                    </div>
                </div>
                <div class="user-dropdown-divider"></div>
                <button class="user-dropdown-item login-btn" onclick="window.location.href='login.html'">
                    <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" style="width:18px;height:18px;"><path d="M15 3h4a2 2 0 0 1 2 2v14a2 2 0 0 1-2 2h-4"></path><polyline points="10 17 15 12 10 7"></polyline><line x1="15" y1="12" x2="3" y2="12"></line></svg>
                    Log in / Sign up
                </button>
            `;
        }
    } else {
        let user = { email: 'user@example.com' };
        try {
            const res = await fetch('http://localhost:8080/auth/me', {
                headers: { 'Authorization': `Bearer ${token}` }
            });
            if (res.ok) {
                user = await res.json();
                const displayName = user.name || user.username || 'User';
                if (userNameEl) userNameEl.textContent = `Welcome ${displayName}!`;
                if (textAvatar) textAvatar.textContent = displayName.substring(0, 2).toUpperCase();

                // Fill profile form fields if logged in
                const avatarLarge = document.getElementById('avatarLarge');
                if (avatarLarge) avatarLarge.textContent = displayName.substring(0, 2).toUpperCase();
                const fieldName = document.getElementById('fieldName');
                if (fieldName) fieldName.value = user.name || '';
                const fieldUsername = document.getElementById('fieldUsername');
                if (fieldUsername) fieldUsername.value = user.username || '';
                const fieldEmail = document.getElementById('fieldEmail');
                if (fieldEmail) fieldEmail.value = user.email || '';
            } else {
                throw new Error('Not authenticated');
            }
        } catch(e) {
            if (userNameEl) userNameEl.textContent = 'User';
        }

        if (userDropdownMenu) {
            userDropdownMenu.innerHTML = `
                <div class="user-dropdown-header">
                    <div class="avatar">${(user.name || user.username || "US").substring(0, 2).toUpperCase()}</div>
                    <div class="user-dropdown-header-info">
                        <div class="user-dropdown-name">
                            ${user.name || user.username || 'User'} <span class="user-dropdown-badge">PRO</span>
                        </div>
                        <div class="user-dropdown-email">${user.email || 'user@example.com'}</div>
                    </div>
                </div>
                <div class="user-dropdown-divider"></div>
                <button class="user-dropdown-item" onclick="window.location.href='index.html'">
                    <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><rect x="3" y="3" width="7" height="7"></rect><rect x="14" y="3" width="7" height="7"></rect><rect x="14" y="14" width="7" height="7"></rect><rect x="3" y="14" width="7" height="7"></rect></svg>
                    Dashboard
                </button>
                <button class="user-dropdown-item" onclick="document.body.classList.toggle('dark-theme')">
                    <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M21 12.79A9 9 0 1 1 11.21 3 7 7 0 0 0 21 12.79z"></path></svg>
                    Dark Mode
                </button>
                <div class="user-dropdown-divider"></div>
                <button class="user-dropdown-item" id="btnDropdownLogout">
                    <svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M9 21H5a2 2 0 0 1-2-2V5a2 2 0 0 1 2-2h4"></path><polyline points="16 17 21 12 16 7"></polyline><line x1="21" y1="12" x2="9" y2="12"></line></svg>
                    Sign Out
                </button>
            `;
            document.getElementById('btnDropdownLogout').addEventListener('click', (e) => {
                e.preventDefault();
                localStorage.removeItem('token');
                window.location.href = 'login.html';
            });
        }
    }
